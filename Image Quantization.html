from sklearn.cluster import KMeans
In [9]:
km4=KMeans(16,random_state=0)
km4.fit(img_data)
ri4=np.zeros(img.shape)
for i in range(img.shape[0]):
  t=km4.predict(img[i])
  ri4[i,:] = [km4.cluster_centers_[l] for l in t]
cv2_imshow(ri4)
t1=cv2.imwrite('/content/drive/MyDrive/Sample Images/4bit Reconstructed.png',ri4)
